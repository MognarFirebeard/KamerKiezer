-- Tabel voor de leerlingen
CREATE TABLE IF NOT EXISTS leerlingen (
    id SERIAL PRIMARY KEY,
    naam VARCHAR(100) NOT NULL,
    geslacht VARCHAR(1) NOT NULL, -- 'M' of 'V'
    unieke_code VARCHAR(10) UNIQUE NOT NULL,
    kamer_id INTEGER DEFAULT NULL -- Wordt later gekoppeld aan een kamer
);

-- Tabel voor de kamers
CREATE TABLE IF NOT EXISTS kamers (
    id SERIAL PRIMARY KEY,
    kamer_nummer VARCHAR(10) UNIQUE NOT NULL,
    capaciteit INTEGER NOT NULL DEFAULT 4,
    geslacht_lock VARCHAR(1) DEFAULT NULL -- 'M', 'V', of NULL (indien leeg)
);
